<script>
    const player = document.getElementById('player');
    const gameContainer = document.getElementById('game-container');
    const princesa = document.getElementById('princesa');
    const telaInicio = document.getElementById('tela-inicio');
    const telaFinal = document.getElementById('tela-final');
    const telaCelebracao = document.getElementById('tela-celebracao');
    const btnNao = document.getElementById('btn-nao');

    let playerY = 80;
    let isJumping = false;
    let velocityY = 0;
    let gamePos = 0;
    let gameActive = false;
    const finalDaFase = 2700;

    // Inicializa posição do botão não
    btnNao.style.left = 'calc(50% + 20px)';
    btnNao.style.top = '0px';

    function iniciarJogo(event) {
        event.stopPropagation();
        telaInicio.style.display = 'none';
        gameContainer.style.display = 'block';
        gameActive = true;
        loop(); // Inicia o loop do jogo
    }

    function pular() {
        if (!isJumping && gameActive) {
            isJumping = true;
            velocityY = -15;
        }
    }

    function loop() {
        if (gameActive) {
            // Física do Pulo
            velocityY += 0.8;
            playerY -= velocityY;
            if (playerY <= 80) { playerY = 80; isJumping = false; velocityY = 0; }
            player.style.bottom = playerY + 'px';

            // Movimento do Fundo
            gamePos += 3;
            gameContainer.style.backgroundPosition = `-${gamePos}px bottom`;
            
            // Movimento dos Obstáculos
            document.querySelectorAll('.obstaculo').forEach(obs => {
                let left = parseInt(obs.style.left) - 3;
                obs.style.left = left + 'px';
            });

            // Aproximação da Princesa
            if (gamePos > finalDaFase - 500) {
                princesa.style.right = (parseInt(princesa.style.right || -150) + 3) + 'px';
                if (gamePos > finalDaFase - 100) {
                    gameActive = false;
                    telaFinal.style.display = 'flex';
                }
            }
            requestAnimationFrame(loop);
        }
    }

    function fugir(event) {
        if(event) event.stopPropagation();
        const larguraArea = window.innerWidth - 120;
        const alturaArea = window.innerHeight - 120;
        btnNao.style.position = 'fixed';
        btnNao.style.left = Math.random() * larguraArea + 'px';
        btnNao.style.top = Math.random() * alturaArea + 'px';
    }

    function aceitou(event) {
        event.stopPropagation();
        telaFinal.style.display = 'none';
        telaCelebracao.style.display = 'flex';
    }
</script>
